1 /** The allplayers namespace. */
  2 var allplayers = allplayers || {};
  3 
  4 (function($) {
  5 
  6   /**
  7    * @class The date class wraps up the AllPlayers Date-Time object used in
  8    * several parameters for Event creation, etc.
  9    *
 10    * <p><strong>Usage:</strong></p>
 11    * <pre><code>
 12    *   var start = new Date('2010-09-01T00:00:00');  // Start on 9-1-2011
 13    *   var end = new Date('2012-09-20T00:00:00');    // End on 9-20-2012
 14    *   var repeat = {
 15    *     interval:1,                            // The repeat interval.
 16    *     freq:'DAILY',                          // Repeat Daily
 17    *     until:new Date('2012-09-04T00:00:00'), // Go until 9-4-2012
 18    *     bymonth: ['1'],                        // January
 19    *     bymonthday: ['3'],                     // 3rd of the month.
 20    *     byday: [
 21    *       'SU' => 'SU',                        // Sunday
 22    *       '+1MO' => '+1MO',                    // First Monday of the Month
 23    *       '+2WED' => '+2WED',                  // 2nd Wed of the month
 24    *     ],
 25    *     exdate: [
 26    *       '2011-09-04T00:00:00'                // Except 9-4-2011
 27    *       '2011-10-03T00:00:00'                // Except 10-3-2011
 28    *     ],
 29    *     rdate: [
 30    *       '2011-09-01T00:00:00'                // Add 9-1-2011
 31    *       '2011-10-04T00:00:00'                // ADD 10-4-2011
 32    *     ]
 33    *   };
 34    *
 35    *   // Create a new AllPlayers Date object.
 36    *   var date = new allplayers.date(start, end, repeat);
 37    *
 38    *   // Add additional exceptions.
 39    *   date.addException('2011-09-10T00:00:00');
 40    *
 41    *   // Add additional dates.
 42    *   date.addRDate('2011-10-10T00:00:00');
 43    * </code></pre>
 44    *
 45    * @param {Date} start The start date.
 46    * @param {Date} end The end date.
 47    * @param {object} repeat The repeat rule. In the following form.
 48    */
 49   allplayers.date = function(start, end, repeat) {
 50 
 51     /**
 52      * Creates a new date based on a parameter which could be a string, Date
 53      * object, or nothing...
 54      *
 55      * @param {optional} date Either a date string, Date object, or nothing...
 56      */
 57     this.newDate = function(date) {
 58       if (typeof date === 'string') {
 59         return new Date(date);
 60       }
 61       else if (typeof date === 'object') {
 62         return date;
 63       }
 64       else {
 65         return new Date();
 66       }
 67     }
 68 
 69     /** The start date */
 70     this.start = this.newDate(start);
 71 
 72     /** The end date */
 73     this.end = this.newDate(end);
 74 
 75     /** The repeat rule */
 76     this.repeat = repeat ? {
 77       interval: (repeat.interval ? repeat.interval : 1),
 78       freq: (repeat.freq ? repeat.freq : 'DAILY'),
 79       until: this.newDate(repeat.until),
 80       bymonth: (repeat.bymonth ? repeat.bymonth : []),
 81       bymonthday: (repeat.bymonthday ? repeat.bymonthday : []),
 82       byday: (repeat.byday ? repeat.byday : []),
 83       exdate: (repeat.exdate ? repeat.exdate : []),
 84       rdate: (repeat.rdate ? repeat.rdate : [])
 85     } : null;
 86   };
 87 
 88   // Need to fix the Date prototype to allow toISOString.
 89   if (!Date.prototype.toISOString) {
 90     function padzero(n) {
 91       return n < 10 ? '0' + n : n;
 92     }
 93     function pad2zeros(n) {
 94       if (n < 100) {
 95         n = '0' + n;
 96       }
 97       if (n < 10) {
 98         n = '0' + n;
 99       }
100       return n;
101     }
102     Date.prototype.toISOString = function() {
103       var ISOString = this.getUTCFullYear() + '-';
104       ISOString += padzero(this.getUTCMonth() + 1) + '-';
105       ISOString += padzero(this.getUTCDate()) + 'T';
106       ISOString += padzero(this.getUTCHours()) + ':';
107       ISOString += padzero(this.getUTCMinutes()) + ':';
108       ISOString += padzero(this.getUTCSeconds()) + '.';
109       ISOString += pad2zeros(this.getUTCMilliseconds()) + 'Z';
110       return ISOString;
111     };
112   }
113 
114   /**
115    * Updates the date start and end dates and repeat rule.
116    *
117    * @param {Date} start The new start date.
118    * @param {Date} end The new end date.
119    * @param {object} repeat The new repeat rule.
120    */
121   allplayers.date.prototype.update = function(start, end, repeat) {
122     this.start = start ? this.newDate(start) : this.start;
123     this.end = end ? this.newDate(end) : this.end;
124     if (repeat) {
125       repeat.until = this.newDate(repeat.until);
126       $.extend(this.repeat, repeat);
127     }
128   };
129 
130   /**
131    * Adds a generic new date to repeat rule.
132    *
133    * @param {string} param The repeat rule parameter to set.
134    * @param {optional} date Either a date string, Date object, or nothing...
135    */
136   allplayers.date.prototype.addDate = function(param, date) {
137 
138     // Normalize the date parameter.
139     date = this.newDate(date);
140 
141     // Add this date.
142     this.repeat[param].push(date);
143   };
144 
145   /**
146    * Add's an exception date to the repeat rule.
147    *
148    * @param {Date} except An exception date to remove from the repeat rule.
149    */
150   allplayers.date.prototype.addException = function(except) {
151 
152     // Add an exception.
153     this.addDate('except', except);
154   };
155 
156   /**
157    * Adds an additional date to the repeat rule.
158    *
159    * @param {Date} addition An additional date to add to the repeat rule.
160    */
161   allplayers.date.prototype.addRDate = function(addition) {
162 
163     // Add an addition.
164     this.addDate('rdate', addition);
165   };
166 
167   /**
168    * Returns the object which will be passed to the services API.
169    */
170   allplayers.date.prototype.getObject = function() {
171     var i = 0;
172     var obj = {
173       start: this.start.toISOString(),
174       end: this.end.toISOString()
175     };
176 
177     // If there is a repeat rule, then add that to the object.
178     if (this.repeat) {
179       obj.repeat = {
180         interval: this.repeat.interval,
181         freq: this.repeat.freq,
182         until: this.repeat.until.toISOString(),
183         bymonth: this.repeat.bymonth,
184         bymonthday: this.repeat.bymonthday,
185         byday: this.repeat.byday,
186         exdate: [],
187         rdate: []
188       };
189 
190       // Iterate through the exdate and rdate and add the date strings.
191       i = this.repeat.exdate.length;
192       while (i--) {
193         obj.repeat.exdate.push(this.repeat.exdate[i].toISOString());
194       }
195 
196       i = this.repeat.rdate.length;
197       while (i--) {
198         obj.repeat.rdate.push(this.repeat.rdate[i].toISOString());
199       }
200     }
201 
202     return obj;
203   };
204 
205 }(jQuery));
