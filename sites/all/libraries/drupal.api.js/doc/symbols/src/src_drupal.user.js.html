1 // The drupal namespace.
  2 var drupal = drupal || {};
  3 
  4 /** The current logged in user. */
  5 drupal.current_user = null;
  6 
  7 /**
  8  * @constructor
  9  * @extends drupal.entity
 10  * @class The user class
 11  *
 12  * @param {object} object The user object.
 13  * @param {function} callback The function to be called once the user has
 14  * been retrieved from the server.
 15  * @param {object} options Options used to govern functionality.
 16  */
 17 drupal.user = function(object, callback, options) {
 18   drupal.entity.call(this, object, callback, options);
 19 };
 20 
 21 /** Derive from drupal.entity. */
 22 drupal.user.prototype = new drupal.entity();
 23 
 24 /** Reset the constructor. */
 25 drupal.user.prototype.constructor = drupal.user;
 26 
 27 /** Declare the user api. */
 28 drupal.user.api = jQuery.extend(new drupal.api(), {
 29   resource: 'user'
 30 });
 31 
 32 /**
 33  * Returns an index of users.
 34  *
 35  * @param {object} query The query parameters.
 36  * @param {function} callback The callback function.
 37  * @param {object} options Options used to govern functionality.
 38  */
 39 drupal.user.index = function(query, callback, options) {
 40   drupal.entity.index(drupal.user, query, callback, options);
 41 };
 42 
 43 /**
 44  * Sets the object.
 45  *
 46  * @param {object} object The object which contains the data.
 47  */
 48 drupal.user.prototype.set = function(object) {
 49   drupal.entity.prototype.set.call(this, object);
 50 
 51   /** The name of this entity. */
 52   this.entityName = 'user';
 53 
 54   /** Set the api. */
 55   this.api = drupal.user.api;
 56 
 57   /** Set the ID based on the uid. */
 58   this.id = object.uid || this.id || 0;
 59 
 60   // Set the values for this entity.
 61   this.setValues({
 62     name: '',
 63     mail: '',
 64     pass: '',
 65     status: 1
 66   }, object);
 67 };
 68 
 69 /**
 70  * Sets a user session.
 71  *
 72  * @param {string} name The name of the session.
 73  * @param {string} sessid The session ID.
 74  */
 75 drupal.user.prototype.setSession = function(name, sessid) {
 76 
 77   /** Set the session id for this user. */
 78   this.sessid = sessid;
 79 
 80   // Only set the session name if this user is valid and has a session name.
 81   if (this.id && name) {
 82 
 83     /** Set the session name for this user. */
 84     this.session_name = name;
 85 
 86     /** Now store this in a cookie for further authentication. */
 87     drupal.cookie(name, sessid);
 88 
 89     // Now store this user as the 'current' user.
 90     drupal.current_user = this;
 91   }
 92 };
 93 
 94 /**
 95  * Login a user.
 96  *
 97  * @param {function} callback The callback function.
 98  */
 99 drupal.user.prototype.login = function(callback) {
100   if (this.api) {
101     this.api.execute('login', {
102       username: this.name,
103       password: this.pass
104     }, (function(user) {
105       return function(object) {
106 
107         // Update this object.
108         user.update(object.user);
109 
110         // Set the session.
111         user.setSession(object.session_name, object.sessid);
112 
113         if (callback) {
114           callback.call(user, user);
115         }
116       };
117     })(this));
118   }
119 };
120 
121 /**
122  * Register a user.
123  *
124  * @param {function} callback The callback function.
125  */
126 drupal.user.prototype.register = function(callback) {
127   if (this.api) {
128     this.api.execute('register', this.getPOST(), (function(user) {
129       return function(object) {
130         user.update(object, callback);
131       };
132     })(this));
133   }
134 };
135 
136 /**
137  * Logout the user.
138  *
139  * @param {function} callback The callback function.
140  */
141 drupal.user.prototype.logout = function(callback) {
142   if (this.api) {
143     this.api.execute('logout', null, callback);
144   }
145 };
146 
147 /**
148  * Gets a POST object.
149  *
150  * @return {object} The filtered object.
151  */
152 drupal.user.prototype.getPOST = function() {
153 
154   // Add the password to POST's only.
155   var post = drupal.entity.prototype.getPOST.call(this);
156   post.pass = this.pass;
157   return post;
158 };
159 
160 /**
161  * Returns the object to send to Services.
162  *
163  * @return {object} The object to send to the Services endpoint.
164  */
165 drupal.user.prototype.get = function() {
166   return jQuery.extend(drupal.entity.prototype.get.call(this), {
167     name: this.name,
168     mail: this.mail,
169     status: this.status
170   });
171 };
