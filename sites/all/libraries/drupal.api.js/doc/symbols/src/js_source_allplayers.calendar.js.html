1 /** The allplayers namespace. */
  2 var allplayers = allplayers || {};
  3 
  4 (function($) {
  5 
  6   /** The default options. */
  7   var defaults = {
  8     dialog: '#calendar-dialog-form'
  9   };
 10 
 11   // Store all the calendar instances.
 12   allplayers.calendars = {};
 13 
 14   // Add a way to instanciate using jQuery prototype.
 15   if (!$.fn.allplayers_calendar) {
 16     $.fn.allplayers_calendar = function(options) {
 17       return $(this).each(function() {
 18         if (!allplayers.calendars[$(this).selector]) {
 19           new allplayers.calendar($(this), options);
 20         }
 21       });
 22     };
 23   }
 24 
 25   /**
 26    * @class The AllPlayers calendar JavaScript API
 27    *
 28    * <p><strong>Usage:</strong>
 29    * <pre><code>
 30    *
 31    *   // Create a calendar
 32    *   var player = $("#calendar").apcicalendar({
 33    *
 34    *   });
 35    *
 36    * </code></pre>
 37    * </p>
 38    *
 39    * @param {object} context The jQuery context.
 40    * @param {object} options This components options.
 41    */
 42   allplayers.calendar = function(context, options) {
 43 
 44     // Make sure we provide default options...
 45     var _this = this;
 46     options = $.extend(defaults, options, {
 47       header: {
 48         left: 'prev,next today',
 49         center: 'title',
 50         right: 'month,agendaWeek,agendaDay'
 51       },
 52       editable: true,
 53       dayClick: function(date, allDay, jsEvent, view) {
 54         console.log(date);
 55         console.log(allDay);
 56         console.log(jsEvent);
 57         console.log(view);
 58       },
 59       eventClick: function(event, jsEvent, view) {
 60         console.log(event);
 61         console.log(jsEvent);
 62         console.log(view);
 63         //_this.dialog.show().dialog();
 64       },
 65       eventDrop: function(event, jsEvent, ui, view) {
 66 
 67         // Save this event.
 68         event.update(event);
 69         event.save();
 70       },
 71       eventResizeStop: function(event, jsEvent, ui, view) {
 72 
 73         // Save this event.
 74         event.update(event);
 75         event.save();
 76       },
 77       events: function(start, end, callback) {
 78         _this.getEvents(start, end, callback);
 79       }
 80     });
 81 
 82     /** The calendar dialog to edit events */
 83     this.dialog = $(options.dialog, context).hide();
 84 
 85     // Store this player instance.
 86     allplayers.calendars[options.id] = this;
 87 
 88     // TO-DO: MAKE IT SO THAT WE DON'T NEED A GROUP TO GET EVENTS
 89     this.uuid = '';
 90 
 91     // The api.
 92     this.api = new allplayers.api();
 93 
 94     // Create the fullcalendar.
 95     context.fullCalendar(options);
 96   };
 97 
 98   allplayers.calendar.prototype.onEventClick = function() {
 99     console.log('Event has been clicked');
100   };
101 
102   allplayers.calendar.prototype.getUUID = function(callback) {
103     if (this.uuid) {
104       callback.call(this);
105     }
106     else {
107       var _this = this;
108       this.api.searchGroups({search: 'Spring Soccer 2011'}, function(groups) {
109         _this.uuid = groups[0].uuid;
110         callback.call(_this);
111       });
112     }
113   };
114 
115   /**
116    * Get's all the events in this calendar.
117    *
118    * @param {Date} start The start timeframe.
119    * @param {Date} end The end timeframe.
120    * @param {function} callback The callback function to return the events.
121    */
122   allplayers.calendar.prototype.getEvents = function(start, end, callback) {
123 
124     // Format the start and end strings according to the AllPlayers API.
125     var startString = start.getFullYear() + '-';
126     startString += (start.getMonth() + 1) + '-';
127     startString += start.getDate();
128 
129     var endString = end.getFullYear() + '-';
130     endString += (end.getMonth() + 1) + '-';
131     endString += end.getDate();
132 
133     this.getUUID(function() {
134       var _this = this;
135       this.api.getGroupEvents(this.uuid, {
136         start: startString,
137         end: endString,
138         fields: '*',
139         limit: 0,
140         offset: 0
141       }, function(events) {
142 
143         // Iterate through the events and make them allplayers.event's
144         var i = events.length;
145         while (i--) {
146           events[i].id = events[i].uuid;
147           events[i] = new allplayers.event(_this.api, _this.options, events[i]);
148         }
149 
150         // Add this to the events for the calendar.
151         callback(events);
152       });
153     });
154   };
155 
156 }(jQuery));
