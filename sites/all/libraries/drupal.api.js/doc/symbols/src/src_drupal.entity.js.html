  2 var drupal = drupal || {};

  4 /** Determine if we have storage. */
  5 drupal.hasStorage = (typeof(Storage) !== 'undefined');
  6 drupal.hasStorage &= (typeof(JSON) !== 'undefined');

  8 /**
  9  * @constructor
 10  * @class The base entity class to store the data that is common to all
 11  * drupal entities whether it be groups, events, users, etc.
 12  *
 13  * @param {object} object The entity object.
 14  * @param {function} callback The callback function to get the object.
 15  * @param {object} options Options used to govern functionality.
 16  */
 17 drupal.entity = function(object, callback, options) {
 18 
 19   // Set the options.
 20   this.options = jQuery.extend({
 21     store: true,
 22     expires: 3600
 23   }, (typeof options === 'undefined') ? {} : options);
 24 
 25   // If the object is valid, then set it...
 26   if (object) {
 27     this.set(object);
 28   }
 29 
 30   // If the callback is valid, then load it...
 31   if (callback) {
 32     this.load(callback);
 33   }
 34 };

 36 /**
 37  * Returns an index of entities.
 38  *
 39  * @param {object} object The object to create for each entity.
 40  * @param {object} query The query parameters.
 41  * @param {function} callback The callback function.
 42  * @param {object} options Options used to govern functionality.
 43  */
 44 drupal.entity.index = function(object, query, callback, options) {
 45 
 46   // Set the default options.
 47   options = jQuery.extend({
 48     store: false
 49   }, options || {});
 50 
 51   // Don't require a query...
 52   if (typeof query === 'function') {
 53     callback = query;
 54     query = {};
 55   }
 56 
 57   // Get the list of entities.
 58   var instance = new object({});
 59   instance.api.get({}, instance.getQuery(query), function(entities) {
 60     var i = entities.length;
 61     while (i--) {
 62       entities[i] = new object(entities[i], null, options);
 63       entities[i].store();
 64     }
 65     if (callback) {
 66       callback(entities);
 67     }
 68   });
 69 };

 71 /**
 72  * Sets the defaults for an entities properties.
 73  *
 74  * @param {object} defaults The defaults for the properties being set.
 75  * @param {object} object The object used to set the properties.
 76  */
 77 drupal.entity.prototype.setValues = function(defaults, object) {
 78   for (var name in defaults) {
 79     this[name] = object[name] || this[name] || defaults[name];
 80   }
 81 };

 83 /**
 84  * Update an object.
 85  *
 86  * @param {object} object The object which contains the data.
 87  * @param {function} callback The function to call when it is done updating.
 88  */
 89 drupal.entity.prototype.update = function(object, callback) {
 90 
 91   // Set the object.
 92   if (object) {
 93     this.set(object);
 94   }
 95 
 96   // Now store the object.
 97   this.store();
 98 
 99   // Now callback that this object has been updated.
100   if (callback) {
101     callback.call(this, this);
102   }
103 };

105 /**
106  * Stores the object in local storage.
107  */
108 drupal.entity.prototype.store = function() {
109   if (this.id && this.options.store && drupal.hasStorage) {
110 
111     // Get the object.
112     var object = this.get();
113 
114     // Get the key for this object.
115     var key = this.entityName + '-' + this.id;
116 
117     // Set an expiration date for this object.
118     object.expires = (this.options.expires * 1000) + (new Date()).getTime();
119 
120     // Store this object in localStorage.
121     localStorage.setItem(key, JSON.stringify(object));
122   }
123 };
124 
125 /**
126  * Retrieves an object from local storage.
127  *
128  * @return {object} The object in local storage.
129  */
130 drupal.entity.prototype.retrieve = function() {
131   var object = null, key = '', value = '';
132   if (this.id && this.options.store && drupal.hasStorage) {
133 
134     // Get the key for this object.
135     var key = this.entityName + '-' + this.id;
136 
137     // Get it out of localStorage.
138     if (object = JSON.parse(localStorage.getItem(key))) {
139 
140       // Make sure this object hasn't expired.
141       if ((new Date()).getTime() > object.expires) {
142 
143         // Clear it if it has.
144         localStorage.removeItem(key);
145       }
146       else {
147 
148         // Set the object if it was retrieved.
149         this.set(object);
150       }
151     }
152   }
153   return object;
154 };

156 /**
157  * Clears an item out of local storage.
158  */
159 drupal.entity.prototype.clear = function() {
160   if (this.id && drupal.hasStorage) {
161     var object = this.get(), key = '';
162     for (var prop in object) {
163       key = this.entityName + '-' + this.id + '-' + prop;
164       localStorage.removeItem(key);
165     }
166   }
167 };

169 /**
170  * Sets the object.
171  *
172  * @param {object} object The object which contains the data.
173  */
174 drupal.entity.prototype.set = function(object) {
175 
176   /** The API for this entity */
177   this.api = this.api || null;
178 
179   // Set the values for this entity.
180   this.setValues({
181     id: '',
182     uri: ''
183   }, object);
184 
185   /** The name of this entity. */
186   this.entityName = 'entity';
187 };

189 /**

194 drupal.entity.prototype.get = function() {

201 /**
202  * Gets a POST object.
203  *
204  * @return {object} The filtered object.
205  */
206 drupal.entity.prototype.getPOST = function() {
207   var object = this.get();
208   if (!object.id) {
209     delete object.id;
210   }
211   return object;
212 };

224 /**
225  * Loads and object using the drupal.api.
226  *
227  * @param {function} callback The callback function when the object is
228  * retrieved.
229  */
230 drupal.entity.prototype.load = function(callback) {
231 
232   // If this isn't a valid object, then return null...
233   if (!this.id) {
234     callback(null);
235   }
236 
237   // Declare the object to load...
238   var object = null;
239   if (object = this.retrieve()) {
240     this.update(object, callback);
241   }
242   else if (this.api) {
243 
244     // Call the API.
245     this.api.get(this.get(), {}, (function(entity) {
246       return function(object) {
247 
248         // If no object is returned, then return null.
249         if (!object) {
250           callback(null);
251         }
252 
253         // Update the object.
254         entity.update(object, callback);
255       };
256     })(this));
257   }
258 };

260 /**
261  * Saves this entity.
262  *
263  * @param {function} callback The function called once entity is saved.
264  */
265 drupal.entity.prototype.save = function(callback) {
266 
267   // Check to see if the api is valid.
268   if (this.api) {
269 
270     // Call the api.
271     this.api.save(this.getPOST(), (function(entity) {
272       return function(object) {
273         entity.update(object, callback);
274       };
275     })(this));
276   }
277 };

279 /**
280  * Removes an entity
281  *
282  * @param {function} callback The function called once entity is removed.
283  */
284 drupal.entity.prototype.remove = function(callback) {
285 
286   // Only remove if they have an ID.
287   if (this.id) {
288 
289     // Call the API.
290     this.api.remove(this.get(), callback);
291     this.clear();
292   }
293 };
