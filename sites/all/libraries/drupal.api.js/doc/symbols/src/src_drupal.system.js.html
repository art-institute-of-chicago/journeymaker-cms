 15 /**
 16  * Add a way to store cookies.
 17  *
 18  * @param {string} key The key for the cookie.
 19  * @param {string} value The value of the cookie.
 20  * @param {object} options The options for the cookie storage.
 21  * @return {string} The results of the storage.
 22  */
 23 drupal.cookie = function(key, value, options) {
 24 
 25   // key and at least value given, set cookie...
 26   if (arguments.length > 1 &&
 27      (!/Object/.test(Object.prototype.toString.call(value)) ||
 28       value === null ||
 29       value === undefined)) {
 30     options = $.extend({}, options);
 31 
 32     if (value === null || value === undefined) {
 33       options.expires = -1;
 34     }
 35 
 36     if (typeof options.expires === 'number') {
 37       var days = options.expires, t = options.expires = new Date();
 38       t.setDate(t.getDate() + days);
 39     }
 40 
 41     value = String(value);
 42 
 43     // use expires attribute, max-age is not supported by IE
 44     return (document.cookie = [encodeURIComponent(key), '=',
 45     options.raw ? value : encodeURIComponent(value),
 46     options.expires ? '; expires=' + options.expires.toUTCString() : '',
 47     options.path ? '; path=' + options.path : '',
 48     options.domain ? '; domain=' + options.domain : '',
 49     options.secure ? '; secure' : ''].join(''));
 50   }
 51 
 52   // key and possibly options given, get cookie...
 53   options = value || {};
 54   var decode = options.raw ? function(s) {
 55     return s;
 56   } : decodeURIComponent;
 57 
 58   var pairs = document.cookie.split('; ');
 59   for (var i = 0, pair; pair = pairs[i] && pairs[i].split('='); i++) {
 60     if (decode(pair[0]) === key)
 61       return decode(pair[1] || '');
 62   }
 63   return null;
 64 };

 66 /**
 67  * @constructor
 68  * @class The system class
 69  *
 70  * @param {function} callback The function to be called once the system has
 71  * connected.
 72  * @param {object} options Options used to govern functionality.
 73  */
 74 drupal.system = function(callback, options) {
 75   drupal.entity.call(this, {}, callback, options);
 76 };
 77 
 78 /** Derive from entity. */
 79 drupal.system.prototype = new drupal.entity();
 80 
 81 /** Reset the constructor. */
 82 drupal.system.prototype.constructor = drupal.system;
 83 
 84 /** Declare the system api. */
 85 drupal.system.api = jQuery.extend(new drupal.api(), {
 86   resource: 'system'
 87 });
 88 
 89 /**
 90  * Sets the object.
 91  *
 92  * @param {object} object The object which contains the data.
 93  */
 94 drupal.system.prototype.set = function(object) {
 95   drupal.entity.prototype.set.call(this, object);
 96 
 97   /** The name of this entity. */
 98   this.entityName = 'system';
 99 
100   /** Set the api. */
101   this.api = drupal.system.api;
102 
103   /** Set current user. */
104   this.user = new drupal.user(object.user);
105   this.user.setSession(object.session_name, object.sessid);
106 };
107 
108 /**
109  * Returns the object.
110  *
111  * @return {object} The object to send to the Services endpoint.
112  */
113 drupal.system.prototype.get = function() {
114   return jQuery.extend(drupal.entity.prototype.get.call(this), {
115     user: this.user.get()
116   });
117 };
118 
119 /**
120  * Loads the server.
121  *
122  * @param {function} callback The callback function.
123  */
124 drupal.system.prototype.load = function(callback) {
125 
126   // Connect to the server.
127   this.api.execute('connect', null, (function(system) {
128     return function(object) {
129       system.update(object, callback);
130     };
131   })(this));
132 };
133 
134 /**
135  * Get a variable from the server.
136  *
137  * @param {string} name The variable you wish to retrieve.
138  * @param {string} def The default value of the variable.
139  * @param {function} callback The callback function.
140  */
141 drupal.system.prototype.get_variable = function(name, def, callback) {
142   this.api.execute('get_variable', {
143     name: name,
144     'default': def
145   }, callback);
146 };
147 
148 /**
149  * Set a variable on the server.
150  *
151  * @param {string} name The variable you wish to set.
152  * @param {string} value The value of the variable.
153  * @param {function} callback The callback function.
154  */
155 drupal.system.prototype.set_variable = function(name, value, callback) {
156   this.api.execute('set_variable', {
157     name: name,
158     value: value
159   }, callback);
160 };
161 
162 /**
163  * Delete a variable on the server.
164  *
165  * @param {string} name The variable you wish to set.
166  * @param {function} callback The callback function.
167  */
168 drupal.system.prototype.del_variable = function(name, callback) {
169   this.api.execute('del_variable', {
170     name: name
171   }, callback);
172 };
