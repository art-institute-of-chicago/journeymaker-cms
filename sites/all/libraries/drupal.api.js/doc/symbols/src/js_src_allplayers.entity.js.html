1 /** The allplayers namespace. */
  2 var allplayers = allplayers || {};
  3 
  4 (function($) {
  5 
  6   /**
  7    * @class The base entity class to store the data that is common to all
  8    * allplayers entities whether it be groups, events, users, etc.
  9    *
 10    * @param {object} options The options for this class.
 11    */
 12   allplayers.entity = function(object) {
 13 
 14     /** The universally unique identifier */
 15     this.uuid = '';
 16 
 17     /** The title of this entity */
 18     this.title = '';
 19 
 20     /** The description of this entity */
 21     this.description = '';
 22 
 23     // If object is a string, assume it is a UUID and get it.
 24     this.update(object);
 25   };
 26 
 27   /**
 28    * Get's an object from the AllPlayers API.
 29    *
 30    * @param {function} callback The callback function when the object is
 31    * retrieved.
 32    */
 33   allplayers.entity.prototype.get = function(callback) {
 34   };
 35 
 36   /**
 37    * Update the entity data.
 38    *
 39    * @param {object} entity The entity information.
 40    */
 41   allplayers.entity.prototype.update = function(object) {
 42 
 43     // Update the object.
 44     if (object) {
 45 
 46       // Update the params.
 47       for (var param in object) {
 48         if (object.hasOwnProperty(param) && this.hasOwnProperty(param)) {
 49           this[param] = object[param];
 50         }
 51       }
 52     }
 53   };
 54 
 55   /**
 56    * Returns the object to send during PUT's and POST's during a save or add.
 57    *
 58    * @return {object} An object of the data when saving to the server.
 59    */
 60   allplayers.entity.prototype.getObject = function() {
 61     return {
 62       uuid: this.uuid,
 63       title: this.title,
 64       description: this.description
 65     };
 66   };
 67 }(jQuery));
