1 /** The allplayers namespace. */
  2 var allplayers = allplayers || {};
  3 
  4 (function($) {
  5 
  6   /**
  7    * @class The groups class to govern lists of groups.
  8    *
  9    * @extends allplayers.base
 10    * @param {@link allplayers.api} api The API interface.
 11    * @param {object} options The options for this class.
 12    */
 13   allplayers.groups = function(api, options) {
 14 
 15     // Derive from allplayers.base.
 16     allplayers.base.call(this, api, options);
 17   };
 18 
 19   // Define the prototype for all controllers.
 20   allplayers.groups.prototype = new allplayers.base();
 21   allplayers.groups.prototype.constructor = allplayers.groups;
 22 
 23   /**
 24    * Fetch all groups provided a filter.
 25    *
 26    * @param {string} search Search string to filter groups.
 27    * @param {function} callback The callback function for all the groups.
 28    */
 29   allplayers.groups.prototype.getGroups = function(search, callback) {
 30 
 31     this.api.getGroups(search, function(json) {
 32 
 33       // The groups array.
 34       var groups = [];
 35 
 36       // Iterate through all the groups and return the group objects.
 37       var i = json.length;
 38       while (i--) {
 39         groups.push(new allplayers.group(this.options, this.api, json[i]));
 40       }
 41 
 42       // Return all the group objects.
 43       callback(groups);
 44     });
 45   };
 46 
 47   /**
 48    * Returns group information provided a UUID.
 49    *
 50    * @param {string} uuid The univerally unique identifier for this group.
 51    * @param {function} callback The callback to handle the group object.
 52    */
 53   allplayers.groups.prototype.getGroup = function(uuid, callback) {
 54 
 55     this.api.getGroup(uuid, function(groupInfo) {
 56 
 57       // Return a new group object.
 58       return new allplayers.group(this.options, this.api, groupInfo);
 59     });
 60   };
 61 
 62 }(jQuery));
