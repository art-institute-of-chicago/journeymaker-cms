1 /** The allplayers namespace. */
  2 var allplayers = allplayers || {};
  3 
  4 (function($) {
  5 
  6   /**
  7    * A static object that represents the AllPlayers API.
  8    */
  9   allplayers.api = {
 10 
 11     /** The AllPlayers API path */
 12     path: 'http://www.ttidwell.allplayers.com:8080/api/v1/rest',
 13 
 14     /**
 15      * API function to get any results from the AllPlayers API.
 16      *
 17      * @param {string} type The content type returned from the API
 18      * (groups, events, resources, etc).
 19      *
 20      * @param {object} params The additional params for this API.
 21      * <ul>
 22      * <li><strong>uuid</strong> - The universal unique ID.</li>
 23      * <li><strong>filter</strong> - Additional content type filter.</li>
 24      * <li><strong>query</strong> - key-value pairs to add to query string.</li>
 25      * </ul>
 26      *
 27      * @param {function} callback The callback function.
 28      */
 29     get: function(type, params, callback) {
 30       var path = allplayers.api.path + '/' + type;
 31       path += params.uuid ? ('/' + params.uuid) : '';
 32       path += params.filter ? ('/' + params.filter) : '';
 33       path += '.jsonp?';
 34       path += params.query ? (jQuery.param(params.query) + '&') : '';
 35       $.ajax({
 36         url: path,
 37         dataType: 'jsonp',
 38         success: function(data, textStatus) {
 39           if (textStatus == 'success') {
 40             callback(data);
 41           }
 42           else {
 43             console.log('Error: ' + textStatus);
 44           }
 45         },
 46         error: function(xhr, ajaxOptions, thrownError) {
 47           console.log(xhr.responseText);
 48           callback(null);
 49         }
 50       });
 51     },
 52 
 53     /**
 54      * API function to save any object on the AllPlayers server.  If the object
 55      * already has a UUID, then this is a simple update, otherwise it will
 56      * create a new object.
 57      *
 58      * @param {string} type The content type returned from the API
 59      * (groups, events, resources, etc).
 60      *
 61      * @param {object} object The object you wish to update on the server.
 62      * @param {function} callback The function to be called when the entity has
 63      * finished updating.
 64      */
 65     save: function(type, object, callback) {
 66       var path = allplayers.api.path + '/' + type;
 67       path += object.uuid ? ('/' + object.uuid) : '';
 68       path += '.json';
 69       $.ajax({
 70         url: path,
 71         dataType: 'json',
 72         type: (object.uuid) ? 'PUT' : 'POST',
 73         data: object,
 74         success: function(data, textStatus) {
 75           if (textStatus == 'success') {
 76             callback(data);
 77           }
 78           else {
 79             console.log('Error: ' + textStatus);
 80           }
 81         },
 82         error: function(xhr, ajaxOptions, thrownError) {
 83           console.log(xhr.responseText);
 84           callback(null);
 85         }
 86       });
 87     },
 88 
 89     /**
 90      * Get the groups based on a search query.
 91      *
 92      * @see https://www.allplayers.com/api/v1/rest/wadl/describe.xml#search
 93      */
 94     searchGroups: function(query, callback) {
 95       allplayers.api.get('groups', {
 96         query: query
 97       }, callback);
 98     },
 99 
100     /**
101      * Return a group provided a uuid.
102      *
103      * @see https://www.allplayers.com/api/v1/rest/wadl/describe.xml#uuid
104      */
105     getGroup: function(uuid, query, callback) {
106       allplayers.api.get('groups', {
107         uuid: uuid,
108         query: query
109       }, callback);
110     },
111 
112     /**
113      * Saves a group
114      */
115     saveGroup: function(group, callback) {
116       callback(group);
117     },
118 
119     /**
120      * Returns a groups albums provided a uuid.
121      *
122      * @see https://www.allplayers.com/api/v1/rest/wadl/describe.xml#albums
123      */
124     getGroupAlbums: function(uuid, query, callback) {
125       allplayers.api.get('groups', {
126         uuid: uuid,
127         query: query,
128         filter: 'albums'
129       }, callback);
130     },
131 
132 
133     /**
134      * Returns a groups events provided a uuid.
135      *
136      * @see https://www.allplayers.com/api/v1/rest/wadl/describe.xml#events
137      */
138     getGroupEvents: function(uuid, query, callback) {
139       allplayers.api.get('groups', {
140         uuid: uuid,
141         query: query,
142         filter: 'events'
143       }, callback);
144     },
145 
146     /**
147      * Returns a groups members provided a uuid.
148      *
149      * @see https://www.allplayers.com/api/v1/rest/wadl/describe.xml#members
150      */
151     getGroupMembers: function(uuid, query, callback) {
152       allplayers.api.get('groups', {
153         uuid: uuid,
154         query: query,
155         filter: 'members'
156       }, callback);
157     },
158 
159     /**
160      * Returns a groups photos provided a uuid.
161      *
162      * @see https://www.allplayers.com/api/v1/rest/wadl/describe.xml#photos
163      */
164     getGroupPhotos: function(query, callback) {
165       allplayers.api.get('groups', {
166         query: query,
167         filter: 'photos'
168       }, callback);
169     },
170 
171     /**
172      * Returns an event.
173      */
174     getEvent: function(uuid, fields, callback) {
175       allplayers.api.get('events', {
176         uuid: uuid,
177         query: {
178           fields: fields
179         }
180       }, callback);
181     },
182 
183     /**
184      * Saves an event
185      */
186     saveEvent: function(event, callback) {
187       allplayers.api.save('events', event, callback);
188     }
189   };
190 }(jQuery));
