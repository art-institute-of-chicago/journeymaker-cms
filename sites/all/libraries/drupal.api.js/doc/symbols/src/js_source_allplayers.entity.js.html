1 /** The allplayers namespace. */
  2 var allplayers = allplayers || {};
  3 
  4 (function($) {
  5 
  6   /**
  7    * @class The base entity class to store the data that is common to all
  8    * allplayers entities whether it be groups, events, users, etc.
  9    *
 10    * @extends allplayers.base
 11    * @param {@link allplayers.api} api The API interface.
 12    * @param {object} options The options for this class.
 13    */
 14   allplayers.entity = function(api, options) {
 15 
 16     /** The universally unique identifier */
 17     this.uuid = '';
 18 
 19     /** The title of this entity */
 20     this.title = '';
 21 
 22     /** The description of this entity */
 23     this.description = '';
 24 
 25     // Derive from allplayers.base.
 26     allplayers.base.call(this, api, options);
 27   };
 28 
 29   // Create the proper derivation.
 30   allplayers.entity.prototype = new allplayers.base();
 31   allplayers.entity.prototype.constructor = allplayers.entity;
 32 
 33   /**
 34    * Update the entity data.
 35    *
 36    * @param {object} entity The entity information.
 37    */
 38   allplayers.entity.prototype.update = function(entity) {
 39 
 40     // Allow this to update all the parameters based on what was updated.
 41     $.extend(true, this, entity);
 42   };
 43 
 44   /**
 45    * Returns the object to send during PUT's and POST's during a save or add.
 46    *
 47    * @return {object} An object of the data when saving to the server.
 48    */
 49   allplayers.entity.prototype.getObject = function() {
 50     return {
 51       uuid: this.uuid,
 52       title: this.title,
 53       description: this.description
 54     };
 55   };
 56 }(jQuery));
